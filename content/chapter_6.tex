          \m{q}^{\prime} = \m{p} \\

        \text{where} \quad
        \begin{bmatrix}
          \m{M}(\m{q}, \m{p}, t)      & -\boldsymbol{\Phi}_{\m{q}}(\m{q}, t)^{\top} \\

        \end{bmatrix}
        \begin{bmatrix}
          \dot{\m{p}} \\ \boldsymbol{\lambda}
